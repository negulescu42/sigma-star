\section{Supplementary Note 1 --- Safety frontier and aggregate tail bound}
\begin{proposition}[Safety frontier]
For fixed $A>0$, $d>0$, and tolerance $0<\varepsilon<A$, the equation $\mathrm{cert}(A,\sigma,d)=A\,\Ksig(d^2)=\varepsilon$ has the unique solution
\begin{equation}
\sstar(A,\varepsilon,d)=\frac{d}{\sqrt{2\ln(A/\varepsilon)}},
\end{equation}
and the safe set $\{\sigma>0:\mathrm{cert}(A,\sigma,d)\le\varepsilon\}$ is the interval $(0,\sstar]$.
\end{proposition}
\begin{proof}
$\mathrm{cert}$ is continuous and strictly increasing in $\sigma$ on $(0,\infty)$, from $0^+$ to $A>\varepsilon$; solving $A\,e^{-d^2/2\sigma^2}=\varepsilon$ for $\sigma$ gives the stated $\sstar$, and monotonicity gives the interval.
\end{proof}

\caption{\textbf{The certificate.} (a) Aggregate interference $\mathrm{cert}(\sigma)$ is strictly increasing in $\sigma$; the largest $\sigma$ holding it below the budget $\varepsilon$ is the unique frontier $\sstar$ (Proposition~1). (b) At a query, sources split into near (intended) and far (interference); $\sstar$ bounds the far mass by $\varepsilon$.}

\paragraph{Aggregate tail bound.} The single-source frontier extends to the whole far tail. Every far source satisfies $\lVert y-z_i\rVert>d$, so by monotonicity of the Gaussian its weight obeys $\omega_i=\Ksig(\lVert y-z_i\rVert^2)\le\Ksig(d^2)$. Hence $\sum_{i\in\mathcal{F}}\omega_i^2\le\Ksig(d^2)\sum_{i\in\mathcal{F}}\omega_i$, which rearranges to $\sum_{i\in\mathcal{F}}\omega_i\le\Neff\,\Ksig(d^2)$. Therefore
\begin{equation}
\mathrm{Tail}_\sigma(y,d)=\sum_{i\in\mathcal{F}}|v_i|\,\omega_i\le V_{\max}\,\Neff\,\Ksig(d^2)=\mathrm{cert}(A,\sigma,d),
\end{equation}
with $A=V_{\max}\Neff$, using $|v_i|\le V_{\max}$ then the rearranged inequality. The only assumption is $\omega_i\le\Ksig(d^2)$ for far sources, which the near/far partition (A2) guarantees by construction. The bound is tight when all far sources sit exactly at radius $d$: there $\omega_i=\Ksig(d^2)$ for every $i\in\mathcal{F}$, both inequalities become equalities, and the aggregate tail attains $\mathrm{cert}(A,\sigma,d)$. Setting $\sigma=\sstar$ makes this aggregate leakage at most $\varepsilon$.
\section{Supplementary Note 2 --- Conservativeness of one-shot mass measurement}
\begin{proposition}[Conservativeness]
The effective mass $\Neff(\sigma)$ is nondecreasing in $\sigma$. Consequently $\sstar\le\sigma_{\mathrm{pair}}$, the mass measured at $\sigma_{\mathrm{pair}}$ satisfies $A(\sigma_{\mathrm{pair}})\ge A(\sstar)$, and the true tail at the deployed bandwidth obeys $\mathrm{Tail}_{\sstar}(y,d)\le A(\sigma_{\mathrm{pair}})\,K_{\sstar}(d^2)=\varepsilon$.
\end{proposition}
\begin{proof}
The far set $\mathcal{F}$ is fixed by the geometric radius $d$ and does not depend on $\sigma$, so we may differentiate over a fixed index set. Write the far kernel weights as $w_i=\exp(-r_i^2 u)$, $i\in\mathcal{F}$, with $u=1/(2\sigma^2)$, so $\Neff=S_1^2/S_2$ with $S_1=\sum_{i\in\mathcal{F}} w_i>0$, $S_2=\sum_{i\in\mathcal{F}} w_i^2>0$, and each $w_i$ is a strictly decreasing function of $u$. Differentiating with $\partial_u w_i=-r_i^2 w_i$ gives $\partial_u S_1=-\sum_i r_i^2 w_i$ and $\partial_u S_2=-2\sum_i r_i^2 w_i^2$, so $\partial_u\Neff=(2S_1/S_2^2)\,B$ with $B=S_1\sum_i r_i^2 w_i^2-S_2\sum_i r_i^2 w_i$. Expanding this as a double sum over index pairs and symmetrising by averaging the $(i,j)$ and $(j,i)$ terms,
\begin{equation}\label{eq:symm}
B=\tfrac{1}{2}\sum_{i,j}\left(r_j^2-r_i^2\right)\left(w_j-w_i\right)w_i\,w_j.
\end{equation}
In each summand the map $r_k^2\mapsto w_k=e^{-r_k^2 u}$ is strictly decreasing, so $(r_j^2-r_i^2)$ and $(w_j-w_i)$ always carry opposite signs; their product is $\le0$, and with $w_iw_j>0$ every summand is $\le0$. Hence $B\le0$, giving $\partial_u\Neff\le0$; since $u=1/(2\sigma^2)$ is decreasing in $\sigma$, $\Neff$ is nondecreasing in $\sigma$.

For the ordering, note it does \emph{not} depend on the measured value of $A$. The frontier $\sstar(A)=d/\sqrt{2\ln(A/\varepsilon)}$ is a strictly decreasing function of $A$ on $A>\varepsilon$, and $\sigma_{\mathrm{pair}}=\sstar(1)$ by definition. Since the normalisation $V_{\max}=1$ and the bound $\Neff\ge1$ give $A=V_{\max}\Neff\ge1$ for \emph{any} field and \emph{any} measurement bandwidth, we have $\sstar(A)\le\sstar(1)=\sigma_{\mathrm{pair}}$ before $A$ is evaluated---the inequality is structural, not a consequence of the particular measured mass. Measuring the mass at $\sigma_{\mathrm{pair}}\ge\sstar$ therefore samples $\Neff$ at a larger bandwidth than deployment, and monotonicity gives $A(\sigma_{\mathrm{pair}})\ge A(\sstar)$. Substituting this larger mass into the aggregate bound $\mathrm{Tail}_{\sstar}(y,d)\le A(\sstar)\,K_{\sstar}(d^2)\le A(\sigma_{\mathrm{pair}})\,K_{\sstar}(d^2)=\varepsilon$ yields the claim; the one-shot measurement can only over-state the mass, so the certificate holds without iteration.
\end{proof}

\paragraph{The self-consistent bandwidth.} The one-shot value already certifies, so no iteration is required. If one nonetheless wants the exact bandwidth at which the measurement and deployment scales coincide, note that it is the unique root of the \emph{envelope equation} $g(\sigma):=A(\sigma)\,K_\sigma(d^2)=\varepsilon$: the mass $A(\sigma)=V_{\max}\Neff(\sigma)$ is nondecreasing in $\sigma$ (Proposition~\ref{prop:conserv}) and the kernel $K_\sigma(d^2)=e^{-d^2/2\sigma^2}$ is strictly increasing in $\sigma$, so their product $g$ is strictly increasing and crosses $\varepsilon$ exactly once. That root is therefore located by bisection on $g$ to any tolerance. We caution that the naive re-measurement map $A\!\mapsto\!A(\sstar(A))$ is \emph{order-reversing} (a larger measured mass sets a smaller deployed bandwidth, which re-measures a smaller mass), so the usual ``monotone and bounded $\Rightarrow$ convergent'' argument does not apply to it; the guarantee comes from the one-shot bound above, and the exact root---when wanted---from bisecting the monotone envelope. Empirically the re-measurement sequence does settle in six to seven steps, and every intermediate value is already a valid certificate.

\paragraph{General kernels: the monotonicity is not Gaussian.} The proof above used the Gaussian form $w_i=\exp(-r_i^2 u)$ only through the sign of $\partial_u w_i$; the conclusion holds for a whole class of tails. Let the kernel be $K(r/\sigma)$ for a positive, strictly decreasing $K$, and write $\varphi=\ln K$. For any far pair $r_i<r_j$ the weight ratio is $w_j/w_i=K(r_j/\sigma)/K(r_i/\sigma)$, and
\begin{equation}
\frac{d}{d\sigma}\ln\frac{w_j}{w_i}=\frac{1}{\sigma}\left[\psi(r_i/\sigma)-\psi(r_j/\sigma)\right],\qquad \psi(x):=x\,\varphi'(x)=\frac{x\,K'(x)}{K(x)},
\end{equation}
the \emph{elasticity} of $K$. If $\psi$ is non-increasing---equivalently, $\ln K$ is concave in $\ln x$---then $\psi(r_i/\sigma)\ge\psi(r_j/\sigma)$, so every pairwise ratio $w_j/w_i\le1$ rises toward unity as $\sigma$ grows: the far-weight vector becomes more uniform in the majorisation order. Since $\Neff=(\sum w_i)^2/\sum w_i^2$ is symmetric and Schur-concave, it is nondecreasing under this majorisation. Hence:
\begin{proposition}[Conservativeness, general tail]\label{prop:conserv-gen}
If $K$ is positive, strictly decreasing, and $\ln K(x)$ is concave in $\ln x$, then $\Neff(\sigma)$ is nondecreasing in $\sigma$, and the one-shot mass measurement is conservative exactly as in the Gaussian case.
\end{proposition}
\noindent The Gaussian ($\psi=-x^2$), Laplace ($\psi=-x$), stretched-exponential ($\psi=-\tfrac12 x^{1/2}$), Cauchy ($\psi=-2x^2/(1+x^2)$), and inverse-power ($\psi=-p\,x^p/(1+x^p)$) tails all satisfy the elasticity condition, and we confirm zero monotonicity violations for each on the four public datasets over a $30$-point bandwidth grid. The conservativeness is therefore a property of monotone, log-concave-in-log-distance kernels, not of the Gaussian tail.
\section{Supplementary Note 3 --- Decision stability and performance floor}
\begin{theorem}[Decision stability]
If $\sigma\le\sstar$ and the local margin $m(y)>2\varepsilon$, then the deployed prediction equals the local prediction: the far field cannot change the decision.
\end{theorem}
\begin{proof}
Let $a$ be the local top class and $b$ any other. Then $S_a-S_b=(S_a^{\mathrm{near}}-S_b^{\mathrm{near}})+(S_a^{\mathrm{far}}-S_b^{\mathrm{far}})\ge m(y)-2\varepsilon>0$, since each far term is bounded in magnitude by $\varepsilon$. So $a$ remains the deployed argmax.
\end{proof}

\begin{corollary}[Performance floor]
At $\sigma\le\sstar$, deployed accuracy $\ge$ (near-field accuracy) $-\ \Pr[\,m(y)\le 2\varepsilon\,]$, where $\Pr[\,m(y)\le 2\varepsilon\,]$ is the fraction of queries whose local margin does not clear the budget.
\end{corollary}
\begin{proof}
By Theorem~1 every query with $m(y)>2\varepsilon$ has its deployed prediction equal to its local prediction, so it is correct whenever the local prediction is correct. The deployed error can therefore exceed the near-field error only on the set $\{m(y)\le2\varepsilon\}$, whose measure is $\Pr[\,m(y)\le2\varepsilon\,]$; subtracting this fraction from the near-field accuracy gives the stated floor.
\end{proof}

\paragraph{The role of $2\varepsilon$.} The budget $\varepsilon$ bounds the far part of \emph{each} class score, $|S_c^{\mathrm{far}}|\le\varepsilon$. A pairwise score gap $S_a-S_b$ combines two such scores, so its far perturbation is bounded by $2\varepsilon$; this is why the stability threshold is $2\varepsilon$ rather than $\varepsilon$. A query whose near margin fails to clear $2\varepsilon$ is not necessarily misclassified---it is merely \emph{uncertified}, and a deployer may abstain on exactly this set, which is the abstention reading of the same inequality.
\section{Supplementary Note 4 --- Uniform-over-region certificate}
\begin{proposition}[Uniform-over-region certificate]
Let $B(c,\delta)$ be a ball of radius $\delta$ whose points all remain at distance $>d\ge\sigma$ from every far source. Then
\begin{equation}
\sup_{y\in B(c,\delta)}\mathrm{Tail}_\sigma(y,d)\;\le\;\mathrm{Tail}_\sigma(c,d)+L\,\delta,\qquad L=V_{\max}\,|\mathcal{F}|\,\frac{d}{\sigma^2}\,\Ksig(d^2),
\end{equation}
with Lipschitz constant $L$ independent of the ambient dimension $n$. Choosing $\sigma$ so the right-hand side is $\le\varepsilon$ certifies the whole region.
\end{proposition}
\begin{proof}
$\mathrm{Tail}_\sigma$ is a sum over the far set of Gaussians in $y$; each summand $|v_i|\,\Ksig(\rho_i^2)$ with $\rho_i=\lVert y-z_i\rVert$ has gradient of norm $|v_i|\,(\rho_i/\sigma^2)\,\Ksig(\rho_i^2)$. The map $\rho\mapsto(\rho/\sigma^2)\,\Ksig(\rho^2)=(\rho/\sigma^2)e^{-\rho^2/2\sigma^2}$ is decreasing for $\rho\ge\sigma$, so for every far source, where $\rho_i>d\ge\sigma$, the per-source slope is bounded by $|v_i|\,(d/\sigma^2)\,\Ksig(d^2)\le V_{\max}\,(d/\sigma^2)\,\Ksig(d^2)$. Summing over the $|\mathcal{F}|$ far sources and applying the mean-value inequality from $c$ to $y\in B(c,\delta)$ gives the stated $L$.
\end{proof}

\paragraph{Why the far-set count, not the participation ratio.} The Lipschitz constant sums the \emph{gradient magnitudes} of the individual far Gaussians, and a bound on a sum of $|\mathcal{F}|$ per-source slopes cannot be replaced by the participation ratio $\Neff$: $\Neff$ compresses the far weights through a ratio of $\ell_1$ and $\ell_2$ norms, an operation valid for the tail \emph{value} (Note 1) but not for a sum of slopes, where every far source contributes its full gradient. The Lipschitz constant therefore carries the raw count $|\mathcal{F}|$, not $\Neff$. It remains exponentially small through $\Ksig(d^2)$: at $\sigma=\sstar$, $K_{\sstar}(d^2)=\varepsilon/A$, so $L=(|\mathcal{F}|/A)(d\varepsilon/\sstar^2)$, larger than the erroneous constant when $|\mathcal{F}|\gg\Neff$ and comparable when the far set overlaps little.

\paragraph{Practical interpretation.} On the flagship ResNet-18 field at $\varepsilon=0.05$ the certified ball radius is $\delta\approx0.5$ in the standardised feature metric---strictly positive, so the region certificate is non-vacuous, but small, about $1.6\%$ of the median inter-query distance (ResNet-50 gives $\delta\approx1.0$, also $1.6\%$; ViT gives $\delta\approx0.9$, about $2.2\%$). The uniform-over-region certificate is therefore genuine but tight; the per-query certificate of Theorem~1, which needs no such ball, carries the main deployment claims.



\section{Supplementary Note 5 --- Robustness of the certificate}

\paragraph{Is the near/far radius meaningful?} Assumption (A2) asks that the radius $d$ separate ``intended'' from ``interfering'' sources. We test this directly by measuring, at $\sstar$, the kernel-weighted fraction of near-set mass that shares the query's class---the quantity the vote actually uses. It is $0.31$, $0.25$, and $0.47$ on ResNet-18, ResNet-50, and ViT-B/16, against a chance level of $0.01$ for $100$ classes: a $25$--$47\times$ enrichment, so the near set is strongly class-aligned. Two honest qualifications. This is a \emph{plurality}, not a majority---on two of three backbones most individual near sources are other-class---so ``intended sources'' should be read as ``near sources,'' and the decision-equality guarantee of Theorem~1 is equality with the near-field \emph{vote}, which is itself only weakly pure: the certificate guarantees \emph{which} sources decide, not that they decide correctly. Accuracy is a separate, measured outcome (Table~1 of the main paper), not a corollary of the certificate.

\paragraph{Sensitivity to the budget and radius.} The budget $\varepsilon$ and the radius percentile are inputs, so we sweep them: $\varepsilon\in\{0.01,0.05,0.1,0.2\}$ crossed with the $\{5,10,20\}$th distance percentile for $d$. Deployed accuracy varies by only $2.2$ points (ResNet-18), $6.7$ (ResNet-50), and $2.0$ (ViT) across the whole grid. This flatness must be read correctly: over that grid the realised $\sstar$ itself spans only a $1.34$--$1.41\times$ range (in concentrated high-dimensional geometry the distance percentiles are close and $\sqrt{2\ln(A/\varepsilon)}$ moves slowly), so the informative statement is the \emph{conjunction}---across the grid, accuracy is flat \emph{and} the certified tail stays below budget, while a bandwidth only $4$--$6\times$ larger (the distance heuristics) fails catastrophically. The budget is a genuine dial with a mild, monotone effect on the operating point, not a hidden accuracy knob.

\paragraph{Pointwise versus average.} The certificate is derived per query, but $A$ is measured as an \emph{average} $\Neff$ over the deployment queries, so the reported tail/$\varepsilon$ is strictly an average-case quantity; a per-query guarantee requires the tail to hold at \emph{every} query, not on average. We therefore measure the full per-query distribution on the ResNet-18 field ($10{,}000$ held-out queries at $\sstar$): tail/$\varepsilon$ has mean $0.08$, $95$th percentile $0.18$, $99$th percentile $0.20$, and \emph{maximum} $0.24$---not a single query exceeds the budget. The average-$A$ certificate is thus empirically pointwise here, with fourfold headroom at the worst query.

\paragraph{An a-priori certificate by quantile choice.} The average is a convenience, not a requirement: the same closed form delivers a strict per-query certificate if $A$ is chosen as an upper quantile of $\Neff$ rather than its mean. This is not empirical luck but a direct reading of the frontier. Because $\sstar(A)$ is decreasing in $A$ and $\mathrm{Tail}_\sigma(y,d)\le V_{\max}\Neff(y)\,K_\sigma(d^2)$ pointwise, deploying at $\sstar(A_q)$ with $A_q$ the $(1-\alpha)$-quantile of $\Neff$ over the region guarantees the budget at \emph{every} query whose $\Neff(y)\le A_q$---that is, at all but at most an $\alpha$-fraction of the region by construction; taking $A_q=\sup_y\Neff(y)$ ($\alpha=0$) certifies the whole region uniformly. The quantile level is thus a dial on the allowed per-query failure rate, read off the same geometry, at no labels. Empirically the tail is far below this worst case: at $\alpha=0.01$ (the $99$th percentile) the \emph{measured} maximum tail/$\varepsilon$ over $10{,}000$ queries stays inside the budget on all three backbones---$0.19$ on ResNet-18, $0.17$ on ResNet-50, $0.27$ on ViT-B/16, with zero exceedances anywhere---while accuracy is unchanged to within $0.2$ points because $\sstar(A_{q})$ is only about one percent smaller than $\sstar(A_{\mathrm{mean}})$. Even the fully uniform choice $A_{\max}=\sup_y\Neff(y)$ costs essentially nothing: accuracy changes by at most $0.21$ points from the mean-$A$ field on all three backbones (marginally higher, in fact), again with zero exceedances by construction. So the per-query certificate the theory states is deployable a priori across the suite at negligible cost, with the quantile level set to the failure rate a deployment will tolerate.

\paragraph{Drift of the radius $d$.} The conservativeness result (Proposition~2) is stated at fixed near/far radius $d$. Because $d$ is the tenth percentile of pairwise centre distances, as the field densifies from fixed class-conditional distributions $d$ converges to its population quantile at the sample-quantile rate $O_p(n^{-1/2})$ (Bahadur representation); empirically it drifts by only $1.0\%$ across a $40\times$ increase in field size (a log--log rate exponent of $-0.71$, consistent with the $-1/2$ theory), so re-estimating $d$ per stage corrects a vanishing term and the fixed-$d$ theorem is asymptotically exact.

\section{Supplementary Note 6 --- Selective-prediction operating point}
The abstention readout gives a label-free, per-query support signal: a query with no near mass carries no certificate, and the field declines it. To place its operating point quantitatively, we compare against a margin-threshold selective predictor on the same $\sstar$ vote. At the certificate's $27\%$ coverage on ResNet-18 a labelled margin threshold reaches $68\%$ accuracy, against the field's full-coverage $57\%$ ($74\%$ vs $51\%$ on ResNet-50, $93\%$ vs $72\%$ on ViT-B/16). The certified-stable set therefore sits \emph{on} the field's accuracy--coverage curve rather than above it: $\sstar$'s distinguishing property is not a higher accuracy at matched coverage but that its selection rule is label-free and carries an a-priori interference guarantee, where the margin threshold needs labels to set. This is the sense in which $\sstar$ is a single object---one bound on the far field, read three ways---rather than a competitor to labelled selective prediction.

\section{Supplementary Note 7 --- Reproducibility}

\paragraph{Computational specification.} All reported quantities are computed as follows. Features are standardised per dimension ($z$-score using the training mean and standard deviation); distances are Euclidean in that standardised space. The radius $d$ is the $10$th percentile of pairwise centre distances over a $4{,}000$-centre subsample; $A$ and $\Neff$ are the far-mask participation ratio averaged over $2{,}000$ held-out queries measured at $\sigma_{\mathrm{pair}}$; the reported tail/$\varepsilon$ is the \emph{average} far mass over held-out queries divided by $\varepsilon$ (the full per-query distribution, including the maximum, is reported for the flagship field above); the giant-component percentage is the largest connected component as a fraction of $5{,}000$ sampled centres, with an edge whenever two centres lie within $3\sigma$ (this diagnostic is \emph{illustrative} of field locality, not itself the certificate---a larger $\sigma$ mechanically connects more centres, so it visualises the collapse rather than measuring the interference bound). The labelled comparator (``grid'') selects the bandwidth that maximises held-out accuracy over a $30$-point log-spaced grid; it is a grid search on labelled accuracy, not a $k$-fold procedure. The grid spans $[0.5,30]$ for the vision fields and $[0.3,40]$ for the text and modality fields; for the lower-dimensional public datasets it is log-spaced to cover the data scale of each dataset. The accompanying result files for the vision, text, and modality experiments record the exact grid and selected bandwidth for each. The interference budget is $\varepsilon=0.05$ throughout. Every run uses a fixed seed; the five-seed re-samples vary only the drawn centres and queries. Scripts and result files accompany the manuscript.

\paragraph{Growing-field experiments.} The re-certification result (main text Section~5) uses two growth protocols on CIFAR-100 penultimate features. \emph{Density growth} (three backbones: ResNet-18, ResNet-50, ViT-B/16) densifies the field from $3$ to $120$ examples per class over five stages ($300$ to $12{,}000$ centres, all $100$ classes present throughout so the test set is fixed); a labelled grid search fixes $\sigma$ once on the sparse $3$-per-class field, and at each stage we report $\sstar$ (re-measured), that frozen bandwidth, and a per-stage re-tuned oracle grid. \emph{Class-incremental growth} (ResNet-18) instead adds classes over five stages ($10\!\to\!25\!\to\!50\!\to\!75\!\to\!100$) at a fixed $120$ examples per class, restricting the test set to classes seen so far (standard class-incremental protocol); the labelled grid is fixed once on the initial $10$-class field. All bandwidth selection, radius, mass, and tail definitions match the computational specification above ($\varepsilon=0.05$, $d$ at the $10$th distance percentile, $A$ over $3{,}000$ held-out queries). The class-incremental run is a reported \emph{null}: adding classes at fixed density leaves the field geometry essentially unchanged (the radius $d$ drifts by $-3.5\%$ over the full range and $\sstar$ shrinks only from $6.26$ to $5.43$), so the frozen bandwidth stays within budget (tail/$\varepsilon$ from $0.01$ to $0.23$, never violating) and ties $\sstar$ on accuracy---confirming that the re-certification advantage is driven by densification, not by class count. Both scripts and their result files accompany the manuscript.
\section{Supplementary Note 8 --- Independent numerical verification of Propositions 1--2}

\paragraph{Purpose.} The construction rests on two facts: the participation-ratio tail bound (Proposition~1) and the monotonicity of $\Neff$ in $\sigma$ that makes the one-shot measurement conservative (Proposition~2). Because a certificate is only as good as these, we verify them directly on random configurations, independently of the deep-network experiments.

\paragraph{Tail bound (Proposition~1).} Over $20{,}000$ random all-far configurations---each with a radius $d$, a far-set size $|\mathcal{F}|$ drawn in $[5,200]$, radii $r_i>d$, and a bandwidth $\sigma$ spanning a wide range of $d$---we evaluate the worst-case aligned tail $\sum_{i\in\mathcal{F}}\omega_i$ (all $|v_i|=V_{\max}$) against the certificate $\mathrm{cert}(A,\sigma,d)=\Neff\,\Ksig(d^2)$. The ratio never exceeds one (maximum $0.98$, mean $0.76$; zero violations), and Fig.~\ref{fig:verif}a shows \emph{where} the tightening lives: it approaches one only as $\Neff/|\mathcal{F}|\!\to\!1$ (near-uniform weights), and is strictly below the union/count bound $|\mathcal{F}|\,\Ksig(d^2)$ whenever the far weights are unequal---so the participation ratio is not merely valid but strictly tighter than the count it replaces.

\paragraph{Conservativeness (Proposition~2).} Over $3{,}000$ random far-sets evaluated on a $40$-point $\sigma$ grid at $60$-digit precision, $\Neff(\sigma)$ is monotone nondecreasing with zero violations (smallest consecutive increment $2.1\times10^{-49}$, i.e. numerically non-negative); Fig.~\ref{fig:verif}b shows representative curves. Apparent violations at ordinary floating precision are underflow artifacts as $\sigma\!\to\!0$ and vanish at high precision.

\paragraph{A shorter route to Proposition~2.} The monotonicity has a one-line reading worth recording. The far weights $\omega_i=\exp(-r_i^2/2\sigma^2)$ are a tempered family with inverse temperature $\beta=1/2\sigma^2$, and $\Neff=(\sum_i\omega_i)^2/\sum_i\omega_i^2$ is exactly the inverse-Simpson (RĂ©nyi-2) diversity of the normalised weights. Diversity indices are nondecreasing as a distribution is tempered toward uniformity (increasing $\sigma$, decreasing $\beta$, flattens the weights), which is Proposition~2. The explicit differencing proof of Note~2 is self-contained; this identity connects it to the diversity-index literature and makes the direction of the monotonicity immediate.

\section{Supplementary Note 9 --- Attention as a Gaussian kernel field}\label{si:attention}
\paragraph{The reduction.} Single-head softmax attention assigns query $q$ the weight $a_i=\mathrm{softmax}_i(\langle q,k_i\rangle/\tau)$ over keys $\{k_i\}$. The polarisation identity $\langle q,k_i\rangle=\tfrac12(\lVert q\rVert^2+\lVert k_i\rVert^2-\lVert q-k_i\rVert^2)$ gives
\begin{equation}
\exp\big(\langle q,k_i\rangle/\tau\big)=e^{\lVert q\rVert^2/2\tau}\cdot e^{\lVert k_i\rVert^2/2\tau}\cdot e^{-\lVert q-k_i\rVert^2/2\tau}.
\end{equation}
The three factors are, in order, a query-only term $e^{\lVert q\rVert^2/2\tau}$, a per-key source weight $v_i=e^{\lVert k_i\rVert^2/2\tau}$, and the Gaussian kernel $K_\sigma(\lVert q-k_i\rVert^2)$ with $\sigma^2=\tau$. The query-only factor is independent of $i$ and cancels in the softmax normalisation. Hence
\begin{equation}
a_i=\frac{v_i\,K_\sigma(\lVert q-k_i\rVert^2)}{\sum_j v_j\,K_\sigma(\lVert q-k_j\rVert^2)},\qquad v_i=e^{\lVert k_i\rVert^2/2\tau},\quad \sigma=\sqrt{\tau},
\end{equation}
which is exactly the normalised Gaussian kernel field of the main text with source weights $v_i$ and bandwidth $\sigma=\sqrt\tau$. The safety frontier (Proposition~1) and conservativeness (Proposition~2) apply unchanged, with certified temperature $\tau^*=(\sstar)^2=d^2/(2\ln(A/\varepsilon))$ and $A=V_{\max}\Neff$, $V_{\max}=\max_i v_i$.
\paragraph{Scope: bounded versus unbounded weights.} Assumption (A1) sets $V_{\max}=1$; attention's $v_i=e^{\lVert k_i\rVert^2/2\tau}$ are unbounded, so $V_{\max}$ enters the mass $A$ as a measured quantity rather than being normalised away. The interference-tail certificate (Proposition~1, stated in units of $V_{\max}$) is therefore what transfers; the far-set locality readout (Corollary~2), which compares to the full output magnitude, is $\varepsilon$-tight only when the keys are norm-bounded---the case $\lVert k_i\rVert\equiv$ const, where $v_i$ is constant and attention is a \emph{pure} Gaussian field. When key norms vary, $\tau^*$ still bounds the far interference tail but the normalised far-attention mass is looser: measured at $\tau^*$ it is $0.35$ (BERT) and $0.62$ (GPT-2), decreasing in $\tau$ but not to $\varepsilon$ (it is $0.80$--$0.84$ at $4\tau^*$).
\paragraph{Empirical confirmation.} On the real query/key vectors of every sampled head of BERT-base and GPT-2 over long contexts ($\sim\!360$ keys/head), the identity holds to the models' float32 precision ($<10^{-4}$ over all heads; it is an algebraic equality, exact up to floating-point rounding), and at $\tau^*$ the certified far-key tail $\mathrm{Tail}/V_{\max}$ stays below $\varepsilon$ across $\sim\!35{,}000$ queries per model with zero violations, rising $18$--$33\times$ over budget at $4\tau^*$ (Fig.~6 of the main text; script and outputs in the repository).

\section{Supplementary Note 10 --- Prototypical networks: the bounded-weight case}\label{si:prototype}
\paragraph{The reduction.} A prototypical network assigns a query $x$ to class $k$ by $p(k\mid x)=\mathrm{softmax}_k(-\lVert f(x)-c_k\rVert^2/\tau)$, where $f$ is the embedding and $c_k$ the class prototype (the mean support embedding). This is \emph{immediately} a Gaussian kernel field over the prototypes: with $\sigma^2=\tau/2$,
\begin{equation}
p(k\mid x)=\frac{K_\sigma(\lVert f(x)-c_k\rVert^2)}{\sum_j K_\sigma(\lVert f(x)-c_j\rVert^2)},\qquad v_k=1,\ \sigma=\sqrt{\tau/2}.
\end{equation}
No polarisation is needed---the readout is already a squared-distance kernel---and every source weight is $v_k=1$, so assumption (A1) holds with $V_{\max}=1$ \emph{exactly}. Unlike attention, both the interference-tail certificate (Proposition~1) and the full far-set locality readout (Corollary~2), which compares the far mass to the total output, transfer with $\varepsilon$-tightness.
\paragraph{Empirical confirmation.} We train a standard $4$-layer convolutional ProtoNet ($64$-dimensional embedding) on Omniglot ($20$-way episodes) to \SI{91.5}{\percent} held-out accuracy, and certify its prototype fields on $12{,}000$ held-out queries from $60$-way episodes, with $d$ the tenth-percentile query--prototype distance and $A=\Neff$ measured at the single-source scale $\sigma_{\text{pair}}$ (as in Note~9). The identity holds to float32 ($<10^{-6}$); at $\sstar$ the certified far-prototype tail is $0.25\varepsilon$ with zero violations over all $12{,}000$ queries, rising $\sim\!560\times$ over budget at $4\sstar$. Because the weights are bounded, the normalised far mass at $\sstar$ is a genuinely $\varepsilon$-small $0.08$ (against $0.35$--$0.62$ for attention), confirming that the full locality guarantee---not only the interference tail---transfers in the bounded-weight case (Fig.~7 of the main text; script and outputs in the repository).

\section{Supplementary Note 11 --- From a single head to the full attention layer}\label{si:mha}
\paragraph{Setup.} Note~9 certifies a \emph{single} head: at the certified temperature $\tau^*$, the output-relevant far weight for query $q$ is the normalised far mass $m^h=\sum_{i\in\mathcal{F}}a^h_i$, where $a^h_i$ are the softmax weights of head $h$ and $\mathcal{F}$ is the far-key set. A transformer layer runs $H$ heads in parallel and combines them: head $h$ emits $o^h=\sum_i a^h_i\,\mathrm{val}^h_i$ with value vectors $\mathrm{val}^h_i=W_V^h x_i$, and the layer output is $O=W_O\,[\,o^1;\dots;o^H\,]$ for an output projection $W_O$. We propagate the per-head guarantee to $O$.

\paragraph{Proposition 5 (Multi-head layer perturbation).} Let $O$ be the attention-layer output at query $q$ and $\tilde O$ the output recomputed with the far-key set $\mathcal{F}$ deleted (softmax renormalised over the near keys). Then
\begin{equation}
\|O-\tilde O\|_2 \le \|W_O\|_2 \left( \sum_{h=1}^{H} \left( 2\, m^h\, \|W_V^h\|_2\, X \right)^2 \right)^{1/2}, \quad X=\max_i \|x_i\|_2 ,
\end{equation}
with $m^h$ the normalised far mass of head $h$ at $\tau^*$. The bound is computable from the pretrained weights and the per-head far masses alone; no labels and no gradient are required.

\emph{Proof.} Fix head $h$. Deleting $\mathcal{F}$ replaces $o^h=\sum_i a^h_i\mathrm{val}^h_i$ by the near-only average $\tilde o^h=\sum_{i\notin\mathcal{F}}(a^h_i/n^h)\,\mathrm{val}^h_i$ with $n^h=\sum_{i\notin\mathcal{F}}a^h_i=1-m^h$. Writing $\tilde o^h-o^h=\sum_{i\notin\mathcal{F}}a^h_i(\tfrac1{n^h}-1)\mathrm{val}^h_i-\sum_{i\in\mathcal{F}}a^h_i\mathrm{val}^h_i$ and bounding each value vector by $V^h_{\max}:=\max_i\lVert\mathrm{val}^h_i\rVert_2$, the first sum has total weight $n^h(\tfrac1{n^h}-1)=m^h$ and the second has total weight $m^h$, so $\lVert o^h-\tilde o^h\rVert_2\le 2m^hV^h_{\max}$. Since $\mathrm{val}^h_i=W_V^h x_i$, $V^h_{\max}\le\lVert W_V^h\rVert_2\,X$. Stacking the heads, $O-\tilde O=W_O\,[\,o^1-\tilde o^1;\dots;o^H-\tilde o^H\,]$, and $\lVert O-\tilde O\rVert_2\le\lVert W_O\rVert_2\,\lVert[\,o^h-\tilde o^h\,]_h\rVert_2=\lVert W_O\rVert_2(\sum_h\lVert o^h-\tilde o^h\rVert_2^2)^{1/2}$. Substituting the per-head bound gives the statement. $\blacksquare$

\paragraph{Scope: bounded versus loose.} The layer bound inherits the single-head scope. In the bounded-weight regime ($V_{\max}=1$, as for prototypical networks) each $m^h\le\varepsilon$, and the layer perturbation is $\varepsilon$-small up to the fixed weight-norm factor. For attention the far masses are the measured $0.24$--$0.38$ (Note~9), so the bound is a genuine, computable perturbation guarantee but not $\varepsilon$-small: it certifies that the far field moves the layer output by a bounded amount, not that the amount is negligible. This is the honest statement---a label-free, geometry-only far-field bound for the full attention layer---and it is loose by a measured factor, not vacuous.

\paragraph{Empirical confirmation.} On a real \texttt{bert-base-uncased} layer ($H=12$ heads) over long contexts, we compute the bound from the pretrained $W_V^h$ and $W_O$ (spectral norms $\lVert W_O\rVert_2\approx 2.4$--$2.8$, $\lVert W_V^h\rVert_2\approx 1.1$--$1.4$) and the per-head far masses, and compare it to the \emph{actual} $\lVert O-\tilde O\rVert_2$ from literally deleting the far keys and recomputing the layer output. Across four sampled layers ($0,3,6,9$) and all sampled queries the bound holds without exception. The actual far-deletion perturbation is a non-trivial $12$--$28\%$ of the layer-output norm---attention genuinely moves under far-key deletion, as the unbounded weights predict---and the weight-norm bound exceeds it by a median factor of $\approx\!47$--$70\times$ (tightening to $\approx\!12$--$19\times$ if $V^h_{\max}$ is measured rather than bounded by $\lVert W_V^h\rVert_2 X$). The bound is thus valid and conservative; its looseness is the price of a purely weight-computable, geometry-only certificate (script and outputs in the repository).
\section{Supplementary Note 12 --- The R\'enyi/Hill effective-mass family}\label{si:renyi}
\paragraph{Statement.} The aggregate bound of Note~1 uses the participation ratio $\Neff=(\sum\omega_i)^2/\sum\omega_i^2$, which is the Hill number of the far weights at order $q=2$. It is one member of a family. For $q>1$ define the Hill number of the (nonnegative) far weights $\{\omega_i\}_{i\in\mathcal{F}}$,
\begin{equation}
D_q=\Big(\sum_{i\in\mathcal{F}}p_i^{\,q}\Big)^{1/(1-q)},\qquad p_i=\frac{\omega_i}{\sum_{j\in\mathcal{F}}\omega_j},
\end{equation}
with the limits $D_2=\Neff$ (participation ratio) and $D_\infty=\big(\sum_i\omega_i\big)/\max_i\omega_i$.
\paragraph{Proposition 6 (R\'enyi aggregate bound).} For every $q\ge2$ the far tail obeys $\sum_{i\in\mathcal{F}}\omega_i\le D_q\,K_\sigma(d^2)$, and the family is ordered $D_\infty\le\cdots\le D_q\le\cdots\le D_2$, so larger $q$ gives an equal-or-tighter certificate. The tightest, $D_\infty$, yields $\sum_i\omega_i\le \big(\sum_i\omega_i/\max_i\omega_i\big)\max_i\omega_i$, i.e. the elementary bound $\max_i\omega_i\le K_\sigma(d^2)$.
\begin{proof}
Every far source lies beyond $d$, so $\omega_i=K_\sigma(\lVert y-z_i\rVert^2)\le K_\sigma(d^2)=:\kappa$ by monotonicity. Hence $\max_i\omega_i\le\kappa$, and $\sum_i\omega_i=D_\infty\max_i\omega_i\le D_\infty\,\kappa$. The ordering $D_q$ nonincreasing in $q$ is the standard monotonicity of Hill numbers (R\'enyi entropies are nonincreasing in their order), so $D_\infty\le D_q\le D_2$ for $q\ge2$, and $\sum_i\omega_i\le D_q\kappa$ for each such $q$ follows from the $q=\infty$ case together with $D_q\ge D_\infty$. The $q=2$ case is exactly Note~1.
\end{proof}
\paragraph{Certified bandwidth and conservativeness.} Setting $A_q=V_{\max}D_q$ and inverting $A_q K_\sigma(d^2)=\varepsilon$ gives a $q$-indexed frontier $\sstar_q=d/\sqrt{2\ln(A_q/\varepsilon)}$; since $A_q$ decreases in $q$, $\sstar_q$ increases in $q$, so higher orders certify a larger admissible bandwidth. The conservativeness argument of Note~2 applies to each $D_q$: the map $r_k^2\mapsto\omega_k$ is monotone, so as $\sigma$ grows the normalised weights $p_i$ flatten in the majorisation order, and every Hill number $D_q$ (a Schur-concave, symmetric functional of $p$) is nondecreasing in $\sigma$. One measurement at $\sigma_{\mathrm{pair}}$ therefore over-estimates $D_q(\sstar_q)$ for every $q$.
\paragraph{Empirical confirmation.} On the three frozen CIFAR-100 backbones, measured over $10{,}000$ held-out queries at $\varepsilon=0.05$, the family sweeps a genuine tightness--smoothness axis with the certified tail provably in budget throughout (zero queries over $\varepsilon$ at every $q$, zero monotonicity violations over a $10$-point bandwidth grid):
ResNet-18 $A_2=1.22\times10^4$ ($\sstar=5.22$, tail $0.08\varepsilon$) tightens to $A_\infty=5.1\times10^3$ ($\sstar=5.42$, tail $0.21\varepsilon$), a $2.4\times$ mass reduction;
ResNet-50 $A_2=1.18\times10^4\to A_\infty=4.9\times10^3$ ($2.4\times$);
ViT-B/16 $A_2=1.67\times10^4\to A_\infty=9.8\times10^3$ ($1.7\times$). The participation ratio is thus the loosest useful member; the Hill order is a dial that trades a smoother (more averaged) mass for a tighter certificate, recovering roughly half the mass headroom at $q=\infty$ while keeping the tail below budget (script \texttt{renyi\_conformal.py} in the repository).

\section{Supplementary Note 13 --- The normalized (Gibbs) resolution certificate}\label{si:gibbs}
\paragraph{Motivation.} The base certificate (Notes~1--2) bounds the \emph{unnormalised} far tail $\sum_{i\in\mathcal{F}}v_i\omega_i$. Several modern readouts are \emph{normalised} Gibbs fields---softmax attention, prototypical networks, Nadaraya--Watson regression, energy-based models---where the output-relevant quantity is the normalised far mass $m_F=\sum_{i\in\mathcal{F}}p_i$ with $p_i=e^{-E_i/\tau}/\sum_j e^{-E_j/\tau}$. For these, deleting the far set changes numerator and denominator together, and the unnormalised bound does not transfer verbatim. The following certificate bounds $m_F$ directly.
\paragraph{Theorem 2 (Gibbs far-mass certificate).} Let $p_i=e^{-E_i/\tau}/\sum_j e^{-E_j/\tau}$ be a Gibbs field with energies $E_i$, a near anchor at minimal energy $E_0=\min_i E_i$, and a far set $\mathcal{F}=\{i:E_i\ge E_0+\Delta\}$ separated by an energy gap $\Delta>0$. Let $D_q^{\mathcal{F}}$ be the Hill number (Note~12) of the anchor-shifted far weights $\{e^{-(E_i-E_0)/\tau}\}_{i\in\mathcal{F}}$. Then the normalised far mass obeys
\begin{equation}
m_F=\sum_{i\in\mathcal{F}}p_i\ \le\ D_q^{\mathcal{F}}\,e^{-\Delta/\tau},
\end{equation}
and consequently the certified temperature
\begin{equation}
\tau^*=\frac{\Delta}{\ln\!\big(D_q^{\mathcal{F}}/\varepsilon\big)}
\end{equation}
guarantees $m_F\le\varepsilon$ for every $\tau\le\tau^*$.
\begin{proof}
Divide numerator and denominator of each $p_i$ by $e^{-E_0/\tau}$: with $\hat E_i=E_i-E_0\ge0$, $p_i=e^{-\hat E_i/\tau}/Z$ where $Z=\sum_j e^{-\hat E_j/\tau}\ge e^{-\hat E_0/\tau}=1$ (the anchor contributes $\hat E_0=0$). Hence $m_F=\sum_{i\in\mathcal{F}}e^{-\hat E_i/\tau}/Z\le\sum_{i\in\mathcal{F}}e^{-\hat E_i/\tau}$. Write $w_i=e^{-\hat E_i/\tau}$ for $i\in\mathcal{F}$; each $\hat E_i\ge\Delta$, so $\max_{i\in\mathcal{F}}w_i\le e^{-\Delta/\tau}$. By Note~12 applied to the far weights, $\sum_{i\in\mathcal{F}}w_i\le D_q^{\mathcal{F}}\max_{i\in\mathcal{F}}w_i\le D_q^{\mathcal{F}}e^{-\Delta/\tau}$, giving the bound. The map $\tau\mapsto D_q^{\mathcal{F}}e^{-\Delta/\tau}$ is increasing in $\tau$ (with $D_q^{\mathcal{F}}$ measured at the operating field), so setting it equal to $\varepsilon$ gives the threshold $\tau^*$, and any cooler $\tau\le\tau^*$ yields $m_F\le\varepsilon$.
\end{proof}
\paragraph{Two consequences.} (i) The bound has \emph{no source-amplitude factor}: unlike the additive Gaussian view of attention (Note~9), where the source weights $v_i=e^{\lVert k_i\rVert^2/2\tau}$ are unbounded and enter $A$ through $V_{\max}(\tau)$, the Gibbs certificate controls the output-relevant normalised mass directly. Attention, prototypical networks, and normalised kernel regression are therefore \emph{exact} instances: for attention $E_i=-\langle q,k_i\rangle$ and $\Delta$ is the near/far logit gap; for a prototypical network $E_i=\lVert f(x)-c_i\rVert^2$ and $\Delta=d_{\mathrm{far}}^2-r_{\mathrm{near}}^2$. (ii) It supplies the per-head certified far mass $m^h\le\varepsilon$ that the multi-head layer bound (Note~11) requires, closing that note's remaining looseness at temperatures $\tau\le\tau^*$.
\paragraph{Empirical confirmation on real attention.} On the real query/key vectors of BERT-base and GPT-2 over long contexts ($\sim\!360$ keys/head), with the near anchor the top-decile-logit keys and $\mathcal{F}$ the remainder, the bound $m_F\le D_q^{\mathcal{F}}e^{-\Delta/\tau}$ holds with \emph{zero violations} across $\sim\!33{,}000$ query-heads per model at both $q=2$ and $q=\infty$. At the deployment temperature $\tau=\sqrt{d_k}=8$ the measured normalised far mass is $0.27$ (BERT) and $0.24$ (GPT-2); the certified temperature is far cooler, median $\tau^*\approx0.02$, and at $\tau^*$ the measured far mass collapses below $\varepsilon$ (to $<10^{-10}$) with zero budget violations. The certificate is thus sound and informative: a standard transformer runs much warmer than the far-field budget would require, and the Gibbs certificate quantifies exactly how much cooler the temperature must be to make the output provably local (script \texttt{gibbs\_attention.py} in the repository).

\section{Supplementary Note 14 --- Conformal finite-sample query certificate}\label{si:conformal}
\paragraph{Statement.} The mass $A=V_{\max}\Neff$ is a query-average, so $\sstar(A)$ certifies the \emph{mean} tail. A distribution-free guarantee for a \emph{future single query} follows from conformal prediction with no labels. Let $A_1,\dots,A_n$ be the per-query effective masses on an exchangeable calibration set of unlabelled queries, and let $A_{(1)}\le\cdots\le A_{(n)}$ be their order statistics. Fix a miscoverage level $\alpha$ and set $k=\lceil(n+1)(1-\alpha)\rceil$.
\paragraph{Proposition 7 (Conformal query mass).} Let $A_{\mathrm{conf}}=A_{(k)}$. For a fresh query $Y$ drawn exchangeably with the calibration set, $\Pr[A_Y\le A_{\mathrm{conf}}]\ge1-\alpha$. Consequently deploying at $\sigma^*(A_{\mathrm{conf}})=d/\sqrt{2\ln(A_{\mathrm{conf}}/\varepsilon)}$ certifies the far tail of the fresh query, $\mathrm{Tail}_{\sigma^*(A_{\mathrm{conf}})}(Y,d)\le\varepsilon$, with probability at least $1-\alpha$.
\begin{proof}
By exchangeability of $A_1,\dots,A_n,A_Y$, the rank of $A_Y$ among the $n+1$ values is uniform on $\{1,\dots,n+1\}$, so $\Pr[A_Y\le A_{(k)}]\ge k/(n+1)\ge1-\alpha$ for $k=\lceil(n+1)(1-\alpha)\rceil$. On the event $A_Y\le A_{\mathrm{conf}}$, the aggregate bound gives $\mathrm{Tail}(Y,d)\le A_Y K_{\sigma^*(A_{\mathrm{conf}})}(d^2)\le A_{\mathrm{conf}}K_{\sigma^*(A_{\mathrm{conf}})}(d^2)=\varepsilon$ (the frontier is defined so the last equality holds), which occurs with probability at least $1-\alpha$.
\end{proof}
\paragraph{Empirical confirmation.} On the three CIFAR-100 backbones, splitting the held-out queries into an $n$-point calibration set and a disjoint validation set at $\alpha=0.05$, the fresh-query coverage $\Pr[A_Y\le A_{\mathrm{conf}}]$ is $0.947$, $0.943$, and $0.953$---matching the $0.95$ target---and $A_{\mathrm{conf}}$ exceeds the mean mass $A$ by $10$--$17\%$, so the conformal bandwidth is slightly smaller (more conservative) than the mean-certified one; zero validation queries exceed the budget at $\sigma^*(A_{\mathrm{conf}})$. The conformal wrapper thus upgrades the mean certificate to a distribution-free, label-free, finite-sample guarantee for the next query, at the cost of a fraction of the certified bandwidth (script \texttt{renyi\_conformal.py} in the repository).
